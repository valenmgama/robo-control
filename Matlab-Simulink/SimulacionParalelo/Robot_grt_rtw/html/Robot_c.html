555/* Simplified version of numjac.cpp, for use with RTW. */



556void local_numjac( RTWSolverInfo *si, real_T *y, const real_T *Fty, real_T *fac,



557                  real_T *dFdy )



558{



559  /* constants */



560  real_T THRESH = 1e-6;



561  real_T EPS = 2.2e-16;                /* utGetEps(); */



562  real_T BL = pow(EPS, 0.75);



563  real_T BU = pow(EPS, 0.25);



564  real_T FACMIN = pow(EPS, 0.78);



565  real_T FACMAX = 0.1;



566  int_T nx = 58;



567  real_T *x = rtsiGetContStates(si);



568  real_T del;



569  real_T difmax;



570  real_T FdelRowmax;



571  real_T temp;



572  real_T Fdiff;



573  real_T maybe;



574  real_T xscale;



575  real_T fscale;



576  real_T *p;



577  int_T rowmax;



578  int_T i,j;



579  if (x != y)



580    (void) memcpy(x, y,



581                  (uint_T)nx*sizeof(real_T));



582  rtsiSetSolverComputingJacobian(si,true);



583  for (p = dFdy, j = 0; j < nx; j++, p += nx) {



584    /* Select an increment del for a difference approximation to



585       column j of dFdy.  The vector fac accounts for experience



586       gained in previous calls to numjac. */



587    xscale = fabs(x[j]);



588    if (xscale < THRESH)



589      xscale = THRESH;



590    temp = (x[j] + fac[j]*xscale);



591    del = temp - y[j];



592    while (del == 0.0) {



593      if (fac[j] < FACMAX) {



594        fac[j] *= 100.0;



595        if (fac[j] > FACMAX)



596          fac[j] = FACMAX;



597        temp = (x[j] + fac[j]*xscale);



598        del = temp - x[j];



599      } else {



600        del = THRESH;                  /* thresh is nonzero */



601        break;



602      }



603    }



604



605    /* Keep del pointing into region. */



606    if (Fty[j] >= 0.0)



607      del = fabs(del);



608    else



609      del = -fabs(del);



610



611    /* Form a difference approximation to column j of dFdy. */



612    temp = x[j];



613    x[j] += del;



614    Robot_step();



615    rtsiSetdX(si,p);



616    Robot_forcingfunction();



617    x[j] = temp;



618    difmax = 0.0;



619    rowmax = 0;



620    FdelRowmax = p[0];



621    temp = 1.0 / del;



622    for (i = 0; i < nx; i++) {



623      Fdiff = p[i] - Fty[i];



624      maybe = fabs(Fdiff);



625      if (maybe > difmax) {



626        difmax = maybe;



627        rowmax = i;



628        FdelRowmax = p[i];



629      }



630



631      p[i] = temp * Fdiff;



632    }



633



634    /* Adjust fac for next call to numjac. */



635    if (((FdelRowmax != 0.0) && (Fty[rowmax] != 0.0)) || (difmax == 0.0)) {



636      fscale = fabs(FdelRowmax);



637      if (fscale < fabs(Fty[rowmax]))



638        fscale = fabs(Fty[rowmax]);



639      if (difmax <= BL*fscale) {



640        /* The difference is small, so increase the increment. */



641        fac[j] *= 10.0;



642        if (fac[j] > FACMAX)



643          fac[j] = FACMAX;



644      } else if (difmax > BU*fscale) {



645        /* The difference is large, so reduce the increment. */



646        fac[j] *= 0.1;



647        if (fac[j] < FACMIN)



648          fac[j] = FACMIN;



649      }



650    }



651  }



652



653  rtsiSetSolverComputingJacobian(si,false);



654}                                      /* end local_numjac */

660static void rt_ertODEUpdateContinuousStates(RTWSolverInfo *si )



661{

663  static int_T rt_ODE14x_N[4] = { 12, 8, 6, 4 };

679  real_T *dfdx = id->DFDX;



680  real_T *W = id->W;



681  int_T *pivots = id->pivots;



682  real_T *xtmp = id->xtmp;



683  real_T *ztmp = id->ztmp;



684  int_T *Mpattern_ir = rtsiGetSolverMassMatrixIr(si);



685  int_T *Mpattern_jc = rtsiGetSolverMassMatrixJc(si);



686  real_T *M = id->M;



687  int_T col,row,rowidx;



688  int_T *N = &(rt_ODE14x_N[0]);



689  int_T i,j,k,iter;



690  int_T nx = 58;



691  rtsiSetSimTimeStep(si,MINOR_TIME_STEP);



692



693  /* Save the state values at time t in y, we'll use x as ynew. */



694  (void) memcpy(x0, x1,



695                (uint_T)nx*sizeof(real_T));



696



697  /* Assumes that rtsiSetT and ModelOutputs are up-to-date */



698  if (id->isFirstStep) {



699    local_evaluateMassMatrix(si,M );



700    id->isFirstStep = false;



701  }



702



703  /* f0 = f(t,y) */



704  rtsiSetdX(si, f0);



705  Robot_forcingfunction();



706  local_numjac(si,x0,f0,fac,dfdx );



707  for (j = 0; j < order; j++) {



708    real_T *p;



709    real_T hN = h/N[j];



710



711    /* Get the iteration matrix and solution at t0 */



712



713    /* [L,U] = lu(M - hN*J) */



714    (void) memcpy(W, dfdx,



715                  (uint_T)nx*nx*sizeof(real_T));



716    for (p = W, i = 0; i < nx*nx; i++, p++) {



717      *p *= (-hN);



718    }



719



720    for (col = 0, p = W; col < nx; col++, p += nx) {



721      for (rowidx = Mpattern_jc[col]; rowidx < Mpattern_jc[col+1]; rowidx++) {



722        real_T m_row_col = M[rowidx];



723        row = Mpattern_ir[rowidx];



724        p[row] += m_row_col;



725      }



726    }



727



728    rt_lu_real(W, nx,



729               pivots);



730



731    /* First Newton's iteration at t0. */



732    /* rhs = hN*f0 */



733    for (i = 0; i < nx; i++) {



734      Delta[i] = hN*f0[i];



735    }



736



737    /* Delta = (U \ (L \ rhs)) */



738    rt_ForwardSubstitutionRR_Dbl(W, Delta,



739      f1, nx,



740      1, pivots,



741      1);



742    rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1, f1+nx-1,



743      Delta, nx,



744      1, 0);



745



746    /* ytmp = y0 + Delta



747       ztmp = (ytmp-y0)/h



748     */



749    (void) memcpy(x1, x0,



750                  (uint_T)nx*sizeof(real_T));



751    for (i = 0; i < nx; i++) {



752      x1[i] += Delta[i];



753      ztmp[i] = Delta[i]/hN;



754    }



755



756    /* Additional Newton's iterations, if desired.



757       for iter = 2:NewtIter



758       rhs = hN*feval(odefun,tn,ytmp,extraArgs{:}) - M*(ytmp - yn);



759       if statedepM   % only for state dep. Mdel ~= 0



760       Mdel = M - feval(massfun,tn,ytmp);



761       rhs = rhs + Mdel*ztmp*h;



762       end



763       Delta = ( U \ ( L \ rhs ) );



764       ytmp = ytmp + Delta;



765       ztmp = (ytmp - yn)/h



766       end



767     */



768    rtsiSetT(si, t0);



769    rtsiSetdX(si, f1);



770    for (iter = 1; iter < numIter; iter++) {



771      Robot_step();



772      Robot_forcingfunction();



773      for (i = 0; i < nx; i++) {



774        Delta[i] = hN*f1[i];



775        xtmp[i] = x1[i] - x0[i];



776      }



777



778      /* rhs = hN*f(tn,ytmp) - M*(ytmp-yn) */



779      for (col = 0; col < nx; col++) {



780        for (rowidx = Mpattern_jc[col]; rowidx < Mpattern_jc[col+1]; rowidx++) {



781          real_T m_row_col = M[rowidx];



782          row = Mpattern_ir[rowidx];



783          Delta[row] -= m_row_col*xtmp[col];



784        }



785      }



786



787      rt_ForwardSubstitutionRR_Dbl(W, Delta,



788        f1, nx,



789        1, pivots,



790        1);



791      rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1, f1+nx-1,



792        Delta, nx,



793        1, 0);



794



795      /* ytmp = ytmp + delta



796         ztmp = (ytmp - yn)/h



797       */



798      for (i = 0; i < nx; i++) {



799        x1[i] += Delta[i];



800        ztmp[i] = (x1[i] - x0[i])/hN;



801      }



802    }



803



804    /* Steps from t0+hN to t1 -- subintegration of N(j) steps for extrapolation



805       ttmp = t0;



806       for i = 2:N(j)



807       ttmp = ttmp + hN



808       ytmp0 = ytmp;



809       for iter = 1:NewtIter



810       rhs = (ytmp0 - ytmp) + hN*feval(odefun,ttmp,ytmp,extraArgs{:});



811       Delta = ( U \ ( L \ rhs ) );



812       ytmp = ytmp + Delta;



813       end



814       end



815     */



816    for (k = 1; k < N[j]; k++) {



817      t1 = t0 + k*hN;



818      (void) memcpy(x1start, x1,



819                    (uint_T)nx*sizeof(real_T));



820      rtsiSetT(si, t1);



821      rtsiSetdX(si, f1);



822      for (iter = 0; iter < numIter; iter++) {



823        Robot_step();



824        Robot_forcingfunction();



825        if (iter == 0) {



826          for (i = 0; i < nx; i++) {



827            Delta[i] = hN*f1[i];



828          }



829        } else {



830          for (i = 0; i < nx; i++) {



831            Delta[i] = hN*f1[i];



832            xtmp[i] = (x1[i]-x1start[i]);



833          }



834



835          /* rhs = hN*f(tn,ytmp) - M*(ytmp-yn) */



836          for (col = 0; col < nx; col++) {



837            for (rowidx = Mpattern_jc[col]; rowidx < Mpattern_jc[col+1]; rowidx



838                 ++) {



839              real_T m_row_col = M[rowidx];



840              row = Mpattern_ir[rowidx];



841              Delta[row] -= m_row_col*xtmp[col];



842            }



843          }



844        }



845



846        rt_ForwardSubstitutionRR_Dbl(W, Delta,



847          f1, nx,



848          1, pivots,



849          1);



850        rt_BackwardSubstitutionRR_Dbl(W+nx*nx-1, f1+nx-1,



851          Delta, nx,



852          1, 0);



853



854        /* ytmp = ytmp + Delta



855           ztmp = (ytmp - ytmp0)/h



856         */



857        for (i = 0; i < nx; i++) {



858          x1[i] += Delta[i];



859          ztmp[i] = (x1[i] - x1start[i])/hN;



860        }



861      }



862    }



863



864    /* Extrapolate to order j



865       E(:,j) = ytmp



866       for k = j:-1:2



867       coef = N(k-1)/(N(j) - N(k-1))



868       E(:,k-1) = E(:,k) + coef*( E(:,k) - E(:,k-1) )



869       end



870     */



871    (void) memcpy(&(E[nx*j]), x1,



872                  (uint_T)nx*sizeof(real_T));



873    for (k = j; k > 0; k--) {



874      real_T coef = (real_T)(N[k-1]) / (N[j]-N[k-1]);



875      for (i = 0; i < nx; i++) {



876        x1[i] = E[nx*k+i] + coef*(E[nx*k+i] - E[nx*(k-1)+i]);



877      }



878



879      (void) memcpy(&(E[nx*(k-1)]), x1,



880                    (uint_T)nx*sizeof(real_T));



881    }



882  }



883



884  /* x1 = E(:,1); */



885  (void) memcpy(x1, E,



886                (uint_T)nx*sizeof(real_T));



887



888  /* t1 = t0 + h; */



889  rtsiSetT(si,rtsiGetSolverStopTime(si));



890  Robot_step();



891  Robot_projection();



892  rtsiSetSimTimeStep(si,MAJOR_TIME_STEP);



893}
